struct xsurface_disk
{
	char tileMode;
	WORD unk1; //v6
	WORD surfVertCount;
	WORD surfTriCount;
	while(1) //always incremends pos by >= 2 bytes
	{
		WORD rigidVertCount;
		if(!rigidVertCount)
			break;
		WORD localBoneIndex;
	}
	WORD numBlends; //only if vertListCount == 1
	
	surfVert VERTS[surfVertCount];
	WORD INDICES[surfTriCount * 3];
}

struct surfVert_disk //assumed to be converted to XVERTEXINFO_S
{
	float normal[3];
	BYTE RGBA[4];
	float texCoordX;
	float texCoordY;
	float binormal[3];
	float tangent[3];
	
	if(! __SETP__(_AH & 0x44, 0))
		float unk1[12];
	if(vertListCount == 1)
		float offset[3];
	else
		char numWeights;
		WORD blendBoneIndex;
		float offset[3];
		if(numWeights)
			WORD unk2; //boneOffset == unk2 << 6;
			WORD boneWeight;
}
